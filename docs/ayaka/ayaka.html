 70    @property
 71    def help(self):
 72        '''获取当前状态下的帮助，没有找到则返回介绍'''
 73        if self.group.running_app_name == self.name:
 74            helps = []
 75            state = self.state
 76            helps.extend(self.get_helps(state))
 77
 78            while "." in state:
 79                state = state.rsplit(".", 1)[0]
 80                helps.extend(self.get_helps(state))
 81
 82            helps.extend(self.get_helps("*"))
 83
 84            if helps:
 85                return "\n".join(helps)
 86
 87        return self.intro

 89    @property
 90    def all_help(self):
 91        '''获取介绍以及全部状态下的帮助'''
 92        info = self.intro
 93        for k, v in self._help.items():

 96                info += f"\n[{k}]\n{v}"
 97        return info

269    def pack_messages(self, bot_id, messages):
270        '''转换为cqhttp node格式'''
271        data: List[MessageSegment] = []
272        for m in messages:
273            if isinstance(m, MessageSegment) and m.type == "node":
274                data.append(m)
275            else:
276                m = MessageSegment.node_custom(
277                    user_id=bot_id,
278                    nickname="Ayaka Bot",
279                    content=str(m)
280                )
281                data.append(m)
282        return data

284    async def send_many(self, messages):
285        '''发送合并转发消息，消息的类型可以是 List[Message | MessageSegment | str]'''
286        # 分割长消息组（不可超过100条）谨慎起见，使用80作为单元长度
287        div_len = 80
288        div_cnt = ceil(len(messages) / div_len)
289        for i in range(div_cnt):
290            msgs = self.pack_messages(
291                self.bot_id,
292                messages[i*div_len: (i+1)*div_len]
293            )
294            await self.bot.call_api("send_group_forward_msg", group_id=self.group_id, messages=msgs)

296    def t_check(self, bot_id: int, group_id: int):

301            return

303        # 已禁用
304        group = get_group(bot_id, group_id)
305        app = group.get_app(self.name)
306        if not app:
307            logger.warning(f"群聊({group_id}) 已禁用 {self.name}")
308            return
309
310        return bot
